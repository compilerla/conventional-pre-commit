73def is_conventional(input, types=DEFAULT_TYPES, optional_scope=True): 

74 """ 

75 Returns True if input matches Conventional Commits formatting 

76 https://www.conventionalcommits.org 

77 

78 Optionally provide a list of additional custom types. 

79 """ 

80 input = strip_comments(input) 

81 types = conventional_types(types) 

82 pattern = f"^({r_types(types)}){r_scope(optional_scope)}{r_delim()}{r_subject()}{r_body()}" 

83 regex = re.compile(pattern, re.MULTILINE) 

84 

85 result = regex.match(input) 

86 is_valid = bool(result) 

87 if is_valid and result.group("multi") and not result.group("sep"): 

88 is_valid = False 

89 

90 return is_valid

93def has_autosquash_prefix(input): 

94 """ 

95 Returns True if input starts with one of the autosquash prefixes used in git. 

96 See the documentation, please https://git-scm.com/docs/git-rebase. 

97 

98 It doesn't check whether the rest of the input matches Conventional Commits 

99 formatting. 

100 """ 

101 pattern = f"^(({r_autosquash_prefixes()})! ).*$" 

102 regex = re.compile(pattern, re.DOTALL) 

103 

104 return bool(regex.match(input))
